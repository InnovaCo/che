handler(url, data, method)

Обработчик, вызывает событие "pageTransition:init"

 
  handler = (url, data, method) ->
    events.trigger "pageTransition:init", [url, data, method]

  events.bind "pageTransition:success", (data) ->
    events.trigger "pageTransition:stop", data
